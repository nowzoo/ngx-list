import {
  NgxListRecord,
  NgxListSortFn,
  NgxListColumnValueFn
} from './shared';
import get from 'lodash/get';
import sortBy from 'lodash/sortBy';
import isFunction from 'lodash/isFunction';


export class NgxListSort {

  /**
   * A factory that returns an instance of {@Link NgxListSortFn}.
   * @param  fallbackSortColumn   The column to use to sort records
   * that are equal by the the current sortKey.
   * @param  caseInsensitive      Whether to compare string values case-insensitively. Default true.
   * @param  valueFns             A map from column keys to functions that return the value to be sorted.
   */
  static sortFn(
    fallbackSortColumn?: string,
    caseInsensitive = true,
    valueFns: {[key: string]: NgxListColumnValueFn} = {}
  ): NgxListSortFn {
    const fn: NgxListSortFn = (records: NgxListRecord[], sortColumn: string): NgxListRecord[] => {
      const sortFns: NgxListColumnValueFn[] = [];
      if (sortColumn) {
        sortFns.push(NgxListSort.getColumnValueFn(sortColumn, valueFns, caseInsensitive));
      }
      if (fallbackSortColumn && fallbackSortColumn !== sortColumn) {
        sortFns.push(NgxListSort.getColumnValueFn(fallbackSortColumn, valueFns, caseInsensitive));
      }

      const sorted = sortBy(records, sortFns) as  NgxListRecord[];
      return sorted;
    };
    return fn;
  }

  /**
   * A helper method to get the value function for a column.
   */
  static getColumnValueFn(
    k: string,
    valueFns: {[key: string]: NgxListColumnValueFn} = {},
    caseInsensitive: boolean
  ): NgxListColumnValueFn {
    if (isFunction(valueFns[k])) {
      return valueFns[k];
    }
    return (record) => {
      const value = get(record, k);
      if (caseInsensitive && typeof value === 'string') {
        return (value as string).toLowerCase();
      }
      return value;
    };
  }
}
